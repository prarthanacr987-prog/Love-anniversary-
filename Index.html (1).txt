showQuestion();
}else{

function showTasks(){


document.getElementById("questions").classList.add("hidden");
document.getElementById("tasks").classList.remove("hidden");


let html="";


tasks.forEach((task,index)=>{
html += `
<div class="task">
<label>
<input type="checkbox" class="taskCheck">
${task}
</label>
</div>
`;
});


document.getElementById("taskList").innerHTML=html;
}


function checkTasks(){


let checks=document.querySelectorAll(".taskCheck");


let allDone=true;


checks.forEach(c=>{
if(!c.checked) allDone=false;
});


if(!allDone){
alert("Complete all 18 tasks first ❤️");
return;
}


document.getElementById("tasks").classList.add("hidden");
document.getElementById("finalCard").classList.remove("hidden");
